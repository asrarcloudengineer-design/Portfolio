<script>
   // Initialize Lucide Icons
   lucide.createIcons();
   
   // Utility function to convert filename to title
   function filenameToTitle(filename) {
       if (!filename) return "Untitled";
       const nameWithoutExt = filename.substring(0, filename.lastIndexOf('.'));
       return nameWithoutExt.charAt(0).toUpperCase() + nameWithoutExt.slice(1);
   }

   // --- 1. CONFIGURATION ---
   
   // Configuration for the CSS Image Carousel (JPGs)
   const imageFileNames = [
       "rex.jpg", "flx.jpg", "imagine.jpg", "hlct.jpg", "mtrnd.jpg", "a.jpg",
       "b.jpg", "camper.jpg", "cdr.jpg", "veyra.jpg", "mcw.jpg", "vrx.jpg",
       "hlcts.jpg", "dxy.jpg", "star.jpg", "ctdr.jpg", "xoxo.jpg"
   ];
   
   const portfolioItems = imageFileNames.map((name, index) => {
       const categoryMap = [
           "Vehicle", "Concept", "3D Model", "Sketch", "Render", "WIP",
           "WIP", "Auto Design", "Heavy Vehicle", "Auto Design", "Aerospace",
           "Vehicle", "Sketch", "Concept", "Vehicle", "3D Model", "Render"
       ];
       
       return {
           src: "assets/" + name,
           title: filenameToTitle(name),
           category: categoryMap[index] || "Design"
       };
   });

   // Configuration for the Three.js 3D Viewer (GLBs)
   const modelFiles = [
       { name: "drm.glb", title: "DRM Concept" },
       { name: "f1.glb", title: "F1 Racer" },
       { name: "fly.glb", title: "Fly Drone" },
       { name: "f.glb", title: "F-Series Truck" },
       { name: "x.glb", title: "X-Concept SUV" },
       { name: "view.glb", title: "View Mini" }
   ];


   // --- 2. CSS IMAGE CAROUSEL LOGIC ---
   const imageCarousel = document.getElementById('imageCarousel');
   const numberOfCells = portfolioItems.length;
   let cellWidth = 280;
   
   let radius = Math.round( ( cellWidth / 2 ) / Math.tan( Math.PI / numberOfCells ) ) + 100;
   
   if (window.innerWidth < 600) {
       radius = Math.round( ( cellWidth / 2 ) / Math.tan( Math.PI / numberOfCells ) ) + 70;
       if (radius < 250) radius = 250;
   }

   let currentImageAngle = 0;
   const theta = 360 / numberOfCells;

   // Generate Image Cards
   function initializeImageCarousel() {
        if (numberOfCells > 0) {
           portfolioItems.forEach((item, index) => {
               const card = document.createElement('div');
               card.className = 'carousel-card group';
               
               const cellAngle = theta * index;
               card.style.transform = `rotateY(${cellAngle}deg) translateZ(${radius}px)`;

               card.innerHTML = `
                   <div class="relative w-full h-full bg-gray-800">
                       <img src="${item.src}"
                           alt="${item.title}"
                           onerror="handleImageError(this, '${item.src}')"
                           class="w-full h-full object-cover opacity-100">
                       
                       <!-- Removed card-overlay for captions/text -->
                   </div>
               `;
               imageCarousel.appendChild(card);
           });
       } else {
           imageCarousel.innerHTML = `
               <div class="text-center p-8 bg-black/50 rounded-lg border border-white/10 max-w-md mx-auto">
                   <i data-lucide="image-off" class="w-8 h-8 text-red-500 mx-auto mb-4"></i>
                   <p class="text-white font-bold">No Images Defined</p>
                   <p class="text-sm text-gray-400 mt-2">The list of portfolio items is currently empty.</p>
               </div>
           `;
       }

       imageCarousel.style.transform = `translateZ(-${radius}px) rotateY(0deg)`;
   }

   // Image Error Handling (for the CSS Carousel)
   window.handleImageError = function(imgElement, filepath) {
       console.error(`Image failed to load: ${filepath}.`);
       
       const parent = imgElement.parentNode;
       imgElement.remove();
       
       const errorDiv = document.createElement('div');
       errorDiv.className = 'broken-img w-full h-full';
       errorDiv.innerHTML = `
           <i data-lucide="alert-triangle" class="w-8 h-8 text-red-400 mb-2"></i>
           <span class="text-sm text-red-300">Image Error</span>
           <span class="text-xs text-gray-400 pt-1">Missing file: </span>
           <span class="text-xs text-gray-400 break-all">${filepath}</span>
       `;
       parent.prepend(errorDiv);
       lucide.createIcons();
   }

   // Image Carousel Rotation
   window.rotateImageCarousel = function(direction) {
       currentImageAngle += theta * direction * -1;
       imageCarousel.style.transform = `translateZ(-${radius}px) rotateY(${currentImageAngle}deg)`;
   }

   // Image Carousel Resize Handler
   function resizeImageCarousel() {
       if (window.innerWidth < 600) {
           radius = Math.round( ( cellWidth / 2 ) / Math.tan( Math.PI / numberOfCells ) ) + 70;
           if (radius < 250) radius = 250;
       } else {
           radius = Math.round( ( cellWidth / 2 ) / Math.tan( Math.PI / numberOfCells ) ) + 100;
       }
       
       const cards = document.querySelectorAll('.carousel-card');
       cards.forEach((card, index) => {
           const cellAngle = theta * index;
           card.style.transform = `rotateY(${cellAngle}deg) translateZ(${radius}px)`;
       });

       imageCarousel.style.transform = `translateZ(-${radius}px) rotateY(${currentImageAngle}deg)`;
   }


   // --- 3. THREE.JS GLB VIEWER LOGIC ---
   let scene, camera, renderer, controls;
   let loader, currentModel = null;
   let currentModelIndex = 0;

   // DOM Elements for the viewer
   const canvas = document.getElementById('threeCanvas');
   const loaderEl = document.getElementById('loader');
   const modelTitleEl = document.getElementById('modelTitle');
   const prevBtn = document.getElementById('prevBtn');
   const nextBtn = document.getElementById('nextBtn');
   const viewerContainer = document.querySelector('.viewer-container');
   
   // Initial setup function
   function initThree() {
       // Check if modelFiles is empty before initializing
       if (modelFiles.length === 0) {
           modelTitleEl.textContent = "No 3D Models Configured.";
           loaderEl.style.display = 'none';
           prevBtn.disabled = true;
           nextBtn.disabled = true;
           console.warn("3D viewer initialized, but modelFiles array is empty.");
           return;
       }

       // Scene Setup
       scene = new THREE.Scene();
       scene.background = new THREE.Color(0x1a1a1a);

       // Camera Setup
       camera = new THREE.PerspectiveCamera(75, viewerContainer.clientWidth / viewerContainer.clientHeight, 0.1, 1000);
       camera.position.set(0, 1, 4); // Adjusted Z position for a slightly wider initial view

       // Renderer Setup
       renderer = new THREE.WebGLRenderer({ canvas: canvas, antialias: true });
       renderer.setSize(viewerContainer.clientWidth, viewerContainer.clientHeight);
       renderer.setPixelRatio(window.devicePixelRatio);

       // Lighting
       scene.add(new THREE.AmbientLight(0xffffff, 0.5));
       const directionalLight = new THREE.DirectionalLight(0xffffff, 1.5);
       directionalLight.position.set(5, 5, 5);
       scene.add(directionalLight);

       // Controls Setup
       controls = new THREE.OrbitControls(camera, renderer.domElement);
       controls.enableDamping = true;
       controls.dampingFactor = 0.05;
       controls.screenSpacePanning = false;
       controls.minDistance = 0.5; // Loosened constraint (was 1)
       controls.maxDistance = 15; // Loosened constraint (was 10)
       controls.target.set(0, 0, 0);
       controls.update();

       // GLTF Loader
       loader = new THREE.GLTFLoader();

       // Load the first model
       loadModel(currentModelIndex);
       
       // Start the animation loop
       animate();
   }
   
   function loadModel(index) {
       if (index < 0 || index >= modelFiles.length) return;

       // Immediately show loader and update controls
       loaderEl.style.opacity = '1';
       updateModelControls(index);

       // Clear previous model
       if (currentModel) {
           currentModel.traverse((child) => {
               if (child.isMesh) {
                   child.geometry.dispose();
                   if (Array.isArray(child.material)) {
                       child.material.forEach(material => material.dispose());
                   } else {
                       child.material.dispose();
                   }
               }
           });
           scene.remove(currentModel);
           currentModel = null;
       }

       const modelData = modelFiles[index];
       const modelPath = "assets/" + modelData.name;
       modelTitleEl.textContent = `Loading: ${modelData.title}...`;

       loader.load(
           modelPath,
           (gltf) => {
               currentModel = gltf.scene;
               scene.add(currentModel);

               // Auto-scale and center the model
               const box = new THREE.Box3().setFromObject(currentModel);
               const center = box.getCenter(new THREE.Vector3());
               const size = box.getSize(new THREE.Vector3());
               const maxDim = Math.max(size.x, size.y, size.z);
               const scaleFactor = 2 / maxDim;
               
               currentModel.position.sub(center);
               currentModel.scale.set(scaleFactor, scaleFactor, scaleFactor);
               
               // Reset controls target to the center of the scaled model
               controls.target.set(0, 0, 0);
               controls.update();

               loaderEl.style.opacity = '0';
               modelTitleEl.textContent = modelData.title;
               currentModelIndex = index;
               updateModelControls(index);
           },
           (xhr) => {
               const percent = (xhr.loaded / xhr.total * 100).toFixed(0);
               // Update model title with loading percentage
               modelTitleEl.textContent = `Loading: ${modelData.title} (${percent}%)`;
           },
           (error) => {
               console.error('An error occurred while loading the GLB model:', modelPath, error);
               loaderEl.style.opacity = '0';
               modelTitleEl.textContent = `Error loading ${modelData.title}. (File not found)`;
               currentModelIndex = index;
               updateModelControls(index);
           }
       );
   }

   function changeModel(direction) {
       let newIndex = currentModelIndex + direction;
       if (newIndex >= 0 && newIndex < modelFiles.length) {
           loadModel(newIndex);
       }
   }

   function updateModelControls(index) {
       prevBtn.disabled = index <= 0;
       nextBtn.disabled = index >= modelFiles.length - 1;
   }

   // Attach event listeners for controls
   prevBtn.addEventListener('click', () => changeModel(-1));
   nextBtn.addEventListener('click', () => changeModel(1));


   // --- 4. GLOBAL ANIMATION & RESIZE ---
   function animate() {
       requestAnimationFrame(animate);
       if (controls) controls.update();
       if (renderer) renderer.render(scene, camera);
   }

   function onWindowResize() {
       // Resize Image Carousel
       resizeImageCarousel();

       // Resize 3D Viewer
       if (camera && renderer) {
           const width = viewerContainer.clientWidth;
           const height = viewerContainer.clientHeight;
           camera.aspect = width / height;
           camera.updateProjectionMatrix();
           renderer.setSize(width, height);
       }
   }

   window.addEventListener('resize', onWindowResize, false);
   
   // Start all initialization on window load
   window.onload = function() {
       initializeImageCarousel();
       // Check if initThree should run
       if (document.getElementById('threeCanvas')) {
           initThree();
       }
   };
